冒泡法排序

  时间：本学期第13周周日12月9号； 地点：B栋310； 工具：电脑、单片机工具箱。

CODE SEGMENT
ASSUME CS:CODE
ORG 1000H
START: JMP WHATFKS
  ARRAY DB 94,93,90,80,99,85,92,79,82,95
WHATFKS: LEA BX,ARRAY
  MOV DX,0
  MOV CX,9
AGAIN: MOV AL,[BX]
  MOV AH,[BX+1]
  CMP AL,AH
  JBE ABC
  MOV [BX],AH
  MOV [BX+1],AL
  INC DX
ABC: INC BX
  LOOP AGAIN
  CMP DX,0
  JA WHATFKS
  LEA BX,ARRAY
  JMP $
CODE ENDS
  END START

数码管显示

  时间：本学期第14周12月16号； 地点：同上； 工具：同上

CODE SEGMENT
ASSUME CS:CODE
ORG 2333H
START: JMP START0
       PA EQU 0FF20H
       PB EQU 0FF21H
       PC EQU 0FF22H
       BUF DB ?,?,?,?,?,?
       X DB -9
DATA1: DB 0C0H,0F9H,0A4H,0B0H,99H,92H,82H,0F8H,80H,90H,88H,83H,0C6H,0A1H
       DB 86H,8EH,0FFH,0CH,89H,0DEH,0C7H,8CH,0F3H,0BFH,8FH
START0: CALL BUF1
CON1: CALL DISP
      JMP CON1
BUF1: MOV BUF,00H
      MOV BUF+1,00H
      MOV BUF+2,00H
      MOV BUF+3,00H
      MOV AL,X
      AND AL,AL
      MOV BUF+4,00H
      MOV BUF+5,00H
      JE CON1
      JNS PLUS
      MOV BUF+4,0FH
      MOV BUF+5,0FH
      JMP CON1
PLUS: MOV BUF+4,00H
      MOV BUF+5,01H
      RET
DISP: MOV AL,0FFH
      MOV DX,PA
      OUT DX,AL
      MOV CL,0DFH
      MOV BX,OFFSET BUF
DIS1: MOV AL,[BX]
       MOV AH,00H
       PUSH BX
       MOV BX,OFFSET DATA1
       ADD BX,AX
       MOV AL,[BX]
       POP BX
       MOV DX,PB
       OUT DX,AL
       MOV AL,CL
       MOV DX,PA
       OUT DX,AL
       PUSH CX
DIS2:  MOV CX,0FFFFH
       LOOP $
       POP CX
       CMP CL,0FEH
       JZ LX1
       MOV AL,0FEH
       MOV DX,PA
       OUT DX,AL
       INC BX
       ROR CL,1
       JMP DIS1
LX1:   MOV AL,0FEH
       MOV DX,PB
       OUT DX,AL
       RET
CODE ENDS
  END START
